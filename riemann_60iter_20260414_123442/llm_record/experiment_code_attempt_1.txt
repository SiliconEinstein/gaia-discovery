# Python code to test RH by computing zeros of zeta function
import mpmath
import json

# Set high precision for computation
mpmath.mp.dps = 50

try:
    # Test RH by computing first 100 non-trivial zeros
    # and checking if they lie on Re(s) = 1/2
    
    trials = 100
    passed = 0
    max_deviation = 0.0
    counterexample = None
    
    # Compute zeros using mpmath's zetazero function
    # zetazero(n) returns the n-th non-trivial zero
    for n in range(1, trials + 1):
        # Get the n-th zero
        zero = mpmath.zetazero(n)
        
        # Extract real and imaginary parts
        re_part = float(zero.real)
        im_part = float(zero.imag)
        
        # Check if real part equals 1/2
        deviation = abs(re_part - 0.5)
        max_deviation = max(max_deviation, deviation)
        
        # Verify it's actually a zero
        zeta_value = abs(mpmath.zeta(zero))
        
        # Check conditions:
        # 1. Real part should be 1/2 (within numerical tolerance)
        # 2. Should actually be a zero of zeta
        if deviation < 1e-30 and zeta_value < 1e-30:
            passed += 1
        else:
            if counterexample is None:
                counterexample = {
                    "n": n,
                    "zero": {"re": re_part, "im": im_part},
                    "deviation_from_half": deviation,
                    "zeta_magnitude": float(zeta_value)
                }
    
    # Output results
    result = {
        "passed": passed == trials and counterexample is None,
        "trials": trials,
        "max_error": max_deviation,
        "counterexample": counterexample,
        "summary": f"Computed {trials} zeros: all had Re(s) = 1/2 within numerical precision (max deviation = {max_deviation:.2e}). NOTE: This is NOT a genuine test of RH because mpmath.zetazero() ASSUMES zeros lie on critical line in its algorithm. We are verifying algorithmic consistency, not the conjecture itself."
    }
    
    print(json.dumps(result))
    
except Exception as e:
    result = {
        "passed": False,
        "trials": 0,
        "max_error": None,
        "counterexample": {"error": str(e)},
        "summary": f"Error during computation: {str(e)}"
    }
    print(json.dumps(result))
