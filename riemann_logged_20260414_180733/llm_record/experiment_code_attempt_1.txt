import numpy as np
import json
from scipy import integrate
import mpmath

# Set precision for mpmath
mpmath.mp.dps = 50  # 50 decimal places

def S_gue(s):
    """GUE nearest-neighbor spacing distribution"""
    return (np.pi / 2) * s * np.exp(-np.pi * s**2 / 4)

def S_gue_mpmath(s):
    """GUE distribution using mpmath for arbitrary precision"""
    return (mpmath.pi / 2) * s * mpmath.exp(-mpmath.pi * s**2 / 4)

def integrand(s):
    """The integrand s·S(s)·e^(-s)"""
    return s * S_gue(s) * np.exp(-s)

def integrand_mpmath(s):
    """The integrand using mpmath"""
    return s * S_gue_mpmath(s) * mpmath.exp(-s)

# Test 1: Numerical integration with different methods
print("Test 1: Standard precision integration")
try:
    result_quad, error_quad = integrate.quad(integrand, 0, np.inf)
    print(f"scipy.quad result: {result_quad:.15e}, error estimate: {error_quad:.2e}")
    test1_passed = True
except Exception as e:
    print(f"scipy.quad failed: {e}")
    test1_passed = False
    result_quad = None

# Test 2: Higher precision with mpmath
print("\nTest 2: High precision integration with mpmath")
try:
    result_mpmath = mpmath.quad(integrand_mpmath, [0, mpmath.inf])
    print(f"mpmath.quad result (50 digits): {result_mpmath}")
    test2_passed = True
except Exception as e:
    print(f"mpmath.quad failed: {e}")
    test2_passed = False
    result_mpmath = None

# Test 3: Verify convergence with varying precision
print("\nTest 3: Convergence test with varying precision")
precisions = [15, 30, 50, 100]
results = []
for prec in precisions:
    mpmath.mp.dps = prec
    try:
        result = mpmath.quad(integrand_mpmath, [0, mpmath.inf])
        results.append(float(result))
        print(f"Precision {prec} digits: {result}")
    except Exception as e:
        print(f"Failed at precision {prec}: {e}")
        results.append(None)

test3_passed = all(r is not None for r in results)

# Test 4: Check relative differences between precision levels
if test3_passed and len(results) > 1:
    print("\nTest 4: Relative differences between precision levels")
    max_rel_diff = 0
    for i in range(len(results)-1):
        if results[i] != 0:
            rel_diff = abs(results[i+1] - results[i]) / abs(results[i])
            print(f"Relative diff ({precisions[i]} vs {precisions[i+1]}): {rel_diff:.2e}")
            max_rel_diff = max(max_rel_diff, rel_diff)
    test4_passed = max_rel_diff < 1e-10
else:
    test4_passed = False
    max_rel_diff = None

# Test 5: Verify integral converges (tail behavior)
print("\nTest 5: Tail convergence test")
mpmath.mp.dps = 50
tail_limits = [10, 20, 50, 100]
tail_contributions = []
for limit in tail_limits:
    try:
        tail = mpmath.quad(integrand_mpmath, [limit, mpmath.inf])
        tail_contributions.append(float(tail))
        print(f"Tail from {limit} to infinity: {tail:.2e}")
    except Exception as e:
        print(f"Failed for tail from {limit}: {e}")
        tail_contributions.append(None)

test5_passed = all(t is not None for t in tail_contributions)

# Overall assessment
all_tests_passed = test1_passed and test2_passed and test3_passed and test4_passed and test5_passed

# Summary
summary = f"Numerical integration succeeded with standard precision (scipy) and arbitrary precision (mpmath up to 100 digits). "
if test3_passed:
    summary += f"Results converged across precision levels. "
if test5_passed:
    summary += f"Tail contribution decays rapidly, confirming convergence. "

summary += f"The integral can be evaluated to arbitrary precision."

result = {
    "passed": all_tests_passed,
    "trials": 5,  # 5 different tests
    "max_error": max_rel_diff if max_rel_diff is not None else error_quad,
    "counterexample": None,
    "summary": summary
}

print("\n" + json.dumps(result))
