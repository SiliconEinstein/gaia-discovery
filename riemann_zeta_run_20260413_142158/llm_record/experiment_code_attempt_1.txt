import mpmath
import json
import random

mpmath.mp.dps = 30  # 30 decimal places precision

def test_riemann_hypothesis():
    """
    Test the Riemann Hypothesis by:
    1. Computing known zeros on the critical line and verifying they are zeros
    2. Searching for zeros off the critical line in random regions
    3. Verifying that zeros found by general search converge to Re(s) = 1/2
    """
    
    results = {
        'zeros_on_critical_line': [],
        'off_line_searches': [],
        'counterexample': None
    }
    
    # Part 1: Verify known zeros on critical line (first 100 zeros)
    # The first few imaginary parts of zeros are well-known
    known_gamma_approx = [14.134, 21.022, 25.011, 30.425, 32.935, 
                          37.586, 40.919, 43.327, 48.005, 49.774]
    
    zeros_verified = 0
    max_error_on_line = 0
    
    for gamma_approx in known_gamma_approx:
        # Find zero near 1/2 + i*gamma using Newton's method
        try:
            zero = mpmath.findroot(mpmath.zeta, mpmath.mpc(0.5, gamma_approx))
            real_part = float(zero.real)
            error = abs(real_part - 0.5)
            max_error_on_line = max(max_error_on_line, error)
            
            # Verify it's actually a zero
            zeta_val = abs(mpmath.zeta(zero))
            if zeta_val < 1e-20 and error < 1e-10:
                zeros_verified += 1
                results['zeros_on_critical_line'].append({
                    'gamma': float(zero.imag),
                    'real_part': real_part,
                    'deviation': error
                })
        except:
            pass
    
    # Part 2: Search for zeros off the critical line
    # If RH is true, no zeros should exist with Re(s) != 1/2 in critical strip
    off_line_tests = 0
    suspicious_points = 0
    
    for _ in range(100):
        # Random point in critical strip but away from critical line
        sigma = random.uniform(0.01, 0.49) if random.random() < 0.5 else random.uniform(0.51, 0.99)
        t = random.uniform(10, 1000)
        s = mpmath.mpc(sigma, t)
        
        zeta_val = mpmath.zeta(s)
        off_line_tests += 1
        
        # If |zeta(s)| is very small, investigate further
        if abs(zeta_val) < 0.1:
            # Try to find a zero starting from this point
            try:
                found_zero = mpmath.findroot(mpmath.zeta, s, maxsteps=50)
                found_real = float(found_zero.real)
                found_imag = float(found_zero.imag)
                
                # Check if this is in the critical strip
                if 0 < found_real < 1 and found_imag > 0:
                    zeta_at_found = abs(mpmath.zeta(found_zero))
                    if zeta_at_found < 1e-15:
                        deviation = abs(found_real - 0.5)
                        if deviation > 1e-8:
                            # POTENTIAL COUNTEREXAMPLE!
                            results['counterexample'] = {
                                'real_part': found_real,
                                'imag_part': found_imag,
                                'deviation_from_half': deviation,
                                'zeta_value': float(zeta_at_found)
                            }
                            suspicious_points += 1
            except:
                pass
    
    results['off_line_searches'] = {
        'total_tests': off_line_tests,
        'suspicious_points': suspicious_points
    }
    
    # Part 3: Use mpmath's built-in zero finder to get more zeros
    # and verify they all have Re = 1/2
    additional_zeros = 0
    additional_max_error = 0
    
    for n in range(1, 51):  # First 50 zeros
        try:
            zero = mpmath.zetazero(n)
            real_part = float(zero.real)
            error = abs(real_part - 0.5)
            additional_max_error = max(additional_max_error, error)
            if error < 1e-20:
                additional_zeros += 1
        except:
            pass
    
    # Compile final results
    total_zeros_checked = zeros_verified + additional_zeros
    all_on_critical_line = (results['counterexample'] is None) and (additional_max_error < 1e-15)
    
    max_error = max(max_error_on_line, additional_max_error)
    
    output = {
        'passed': all_on_critical_line,
        'trials': total_zeros_checked + off_line_tests,
        'max_error': float(max_error),
        'counterexample': results['counterexample'],
        'summary': f'Verified {total_zeros_checked} zeros all on critical line (max deviation {max_error:.2e}). ' +
                   f'Searched {off_line_tests} off-line points, no counterexamples found. ' +
                   f'Evidence strongly supports RH but cannot prove it.'
    }
    
    print(json.dumps(output))

test_riemann_hypothesis()
